DATA SEGMENT
STR1 DB 100 DUP(?)
STR2 DB 100 DUP(?)
MSG1 DB 10,13."ENTER FIRST STRING$"
MSG2 10,13,"ENTER SECOND STRING$"
MSG3 DB 10,13,"STRINGS ARE EQUAL$"
MSG4 DB 10,13,"STRINGS ARE NOT EQUAL$"
DATA ENDS

DISP MACRO MSG
LEA DX,MSG
MOV AH,09H
INT 21H
ENDM

CODE SEGMENT
ASSUME CS:CODE.DS:DATA
START:MOV AX,DATA
MOV DS,AX

DISPMSG MSG1
MOV SI,00H
MOV CL,00H
UP1:MOV AH,01H
INTY 21H
CMP AL,0DH
JE DN1
MOV STR1[SI],AL
INC SI
INC CL
JMP UP1
DN1:MOV STR1[SI],AL
MOV BL,CL

DISPMSG MSG2
MOV SI,00H
MOV CL,00H
UP2:MOV AH,01H
INT 21H
CMP AL,0DH
JE DNZ
MOV STR2[SI],AL
INC SI
INC CL
JMP UP2
DNZ:MOV STR2[SI],AL

CMP BL,CL
JNE EXIT
LEA SI,STR1
LEA DI,STR2

UP3:MOV AL,[SI]
CMP AL,[DI]

JNE EXIT
INC SI
INC DI
DEC CL
CMP CL,00H
JNE UP3

DISPMSG MSG3
JMP NEXT

EXIT: DISPMSG MSG4
NEXT:MOV AH,04CH
INT 21H

CODE ENDS 
END START
