INCLUDE READ.MAC
INCLUDE DISP.MAC
DATA SEGMENT ARR DB 40 DUP(?)
DATA ENDS

CODE SEGMENT ASSUME CS:CODE,DS:DATA
START :MOV AX,DATA
MOV DS,AX
MOV SI,0
BACK:READ ARR[SI]
INC SI
CMP AL,13
JNZ BACK
MOV DL,10
MOV AH,2
INT 21H
MOV SI,0
AGAIN: DISP ARR[SI]
INC SI
CMP AL,13
JNZ AGAIN
MOV AH,4CH
INT 21H
CODE ENDS
END START

;READ MACRO

READ MACRO VAR
MOV AH,01H
INT 21H
MOV VAR,AL
ENDM

;;DISP.MAC
DISP MACRO VAR
MOV DL,VAR
MOV AH,02H
INT 21H
ENDM
