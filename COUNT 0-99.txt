DISP MACRO MSG
PUSH AX
PUSH DX
LEA DX, MSG
MOV AH.09H
INT 21H
POP DX
POP AX
ENDM
DATA SEGMENT
MSG1 DB 13,"$"
DATA ENDS
CODE SEGMENT
ASSUME CS:CODE,DS:DATA
START: MOV AX,DATA
             MOV DS,AX
             MOV AL,00H
BACK:CALL WRITE
DISP MSG1
;DELAY
MOV CX,0FFFFH
L2:MOV BX,0FFFH
L1:DEC BX
CMP BX,00
JNE L1
LOOP L2
ADD AL,01
DAA
CMP AL,00
JNE BACK
MOV AH,04CH
INT 21H
PUSH AX
MOV BL,AL
MOV DL, AL
MOV CL,04
SHR DL,CL
ADD DL,30H
MOV AH,02H
INT 21H
MOV DL,BL
AND DL,0FH
ADD DL,30H
MOV AH,02H
INT 21H
POP AX
RET
WRITE ENDP
CODE ENDS
END START
