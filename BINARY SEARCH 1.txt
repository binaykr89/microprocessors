DATA SEGMENT
	ARRAY DW 1100H,3322H,6655H,7899H
	LEN DW (LEN-DW)/2
	KEY DW 7899H
	MSG1 DB "FOUND$"
	MSG2 DB"NOT FOUND$"
DATA ENDS

CODE SEGMENT
ASSUME CS:CODE,DS:DATA
START:MOV AX,DATA
MOV DS,AX
MOV BX,00H;START=LOW
MOV DX,LEN;LENGTH
MOV CX,KEY
AGAIN:CMP BX,DX'JA FALIURE;JMP ABOVE
MOV AX,BX
ADD AX,DX
SHR AX,01H;MID
MOV SI,AX
ADD SI,SI
MOV DI,ARRAY[SI]
JAE BIGGER
DEC AX
MOV DX,AX
JMP AGAIN
BIGGER:JE SUCCESS
INC AX
MOV BX,AX
JMP AGAIN
SUCCESS:MOV AH,09H
MOV DX,OFFSET MSG1
INT 21H
JMP DOWN
FAILURE:MOV AH,09H
MOV DX,OFFSET MSG2
INT 21H
DOWN:MOV AH,4CH
INT 21H
CODE ENDS
END START
