DISP MACRO MSG
MOV DX,OFFSET MSG
MOV AH,09H
INT 21H
ENDM
DATA SEGMENT
MSG1 DB 10,13,"ENTER THE FILE NAME TO CREATE:$"
MSG2 DB 10,13,"ENTER FILE NAME TO DELETE:$"
MSG3 DB 10,13,"1: CREATE",10,13,"2:DELETE:$"
MSG4 DB 10,13,"ENTER THE CHOICE$:"
MSG5 DB 10,"ERROR$"
FILENAME DB 80 DUP(?)
DATA ENDS
CODE SEGMENT
ASSUME CS:CODE,DS:DATA
START: MOV AX,DATA
  MOV DS,AX
DISPMSG MSG3
DISPMSG MSG4
MOV AH,01H
INT 21H
CMP AL,'1'
JZ CRTFILE
CMP AL,'2'
JZ DELFILE
ERROR:DISPMSG MSG5
MOV AH,04CH
INT 21H
CRTFILE:DISPMSG MSG1
CALL READ
MOV CX,00
MOV AH,3CH
LEA DX,FILENAME
INT 21H
JC ERROR
MOV AH,04CH
INT 21H
DELFILE:DISPMSG MSG2
CALL READ
LEA DX,FILENAME
MOV AH,41H
INT 21H
JC ERROR
MOV AH,4CH
INT 21H
READ PROC 
MOV AH,01H
LEA SI,FILENAME
BACK:INT 21H
CMP AL,0DH
JZ DONE
MOV [SI],AL
INC SI
JMP BACK
DONE:RET
READ ENDP
CODE ENDS
END START
